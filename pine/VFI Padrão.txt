format_time(ms) =>
    int s = math.floor(ms / 1000)
    int m = math.floor(s / 60)
    int h = math.floor(m / 60)
    s := s % 60
    m := m % 60
    string s_str = s < 10 ? "0" + str.tostring(s) : str.tostring(s)
    string m_str = m < 10 ? "0" + str.tostring(m) : str.tostring(m)
    string h_str = h < 10 ? "0" + str.tostring(h) : str.tostring(h)
    h > 0 ? h_str + ":" + m_str + ":" + s_str : m_str + ":" + s_str

    float tf_mins = timeframe.in_seconds(timeframe.period) / 60
    string tf = "1M"
    if tf_mins < 5 
        tf := "15"
    else if tf_mins >= 5 and tf_mins < 15
        tf := "60"
    else if tf_mins >= 15 and tf_mins < 60
        tf := "240"
    else if tf_mins >= 60 and tf_mins < 240
        tf := "1D"
    else if tf_mins >= 240 and tf_mins < 1440
        tf := "1W"
    else 
        tf := "1M"
    tf

get_cisd_data(bool is_bull) =>
    float trig_p = na
    int trig_t = time
    
    if is_bull 
        bool in_seq = false
        float local_extreme_body = -999999.0
        int local_t = time
        
        for i = 0 to 40
            bool curr_is_bear = close[i] < open[i]
            bool curr_is_bull = close[i] > open[i]
            
            if curr_is_bear
                in_seq := true
                float top_body = math.max(open[i], close[i]) 
                if top_body > local_extreme_body
                    local_extreme_body := top_body
                    local_t := time[i]
            else if curr_is_bull
                if in_seq
                    break 
                    
        if in_seq
            trig_p := local_extreme_body
            trig_t := local_t
    else 
        bool in_seq = false
        float local_extreme_body = 999999999.0
        int local_t = time
        
        for i = 0 to 40
            bool curr_is_bull = close[i] > open[i]
            bool curr_is_bear = close[i] < open[i]
            
            if curr_is_bull
                in_seq := true
                float bot_body = math.min(open[i], close[i]) 
                if bot_body < local_extreme_body
                    local_extreme_body := bot_body
                    local_t := time[i]
            else if curr_is_bear
                if in_seq
                    break 
                    
        if in_seq
            trig_p := local_extreme_body
            trig_t := local_t
            
    if na(trig_p)
        trig_p := is_bull ? math.max(open, close) : math.min(open, close)
        trig_t := time

draw_panel_candle(CandleData c, int start_idx, int w, color c_u_bd, color c_u_br, color c_u_wk, color c_d_bd, color c_d_br, color c_d_wk, bool is_curr) =>
    int x1 = start_idx
    int x2 = start_idx + w
    int cx = int((x1+x2)/2)
    bool is_up = c.c >= c.o
    color col_body   = is_up ? c_u_bd : c_d_bd
    color col_border = is_up ? c_u_br : c_d_br
    color col_wick   = is_up ? c_u_wk : c_d_wk
    color bg = is_curr ? color.new(col_body, 50) : col_body
    array.push(p_boxes, box.new(x1, math.max(c.o, c.c), x2, math.min(c.o, c.c), xloc=xloc.bar_index, border_color=col_border, bgcolor=bg))
    array.push(p_lines, line.new(cx, c.h, cx, math.max(c.o, c.c), xloc=xloc.bar_index, color=col_wick))
    array.push(p_lines, line.new(cx, c.l, cx, math.min(c.o, c.c), xloc=xloc.bar_index, color=col_wick))
    cx

if barstate.islast

    // Select subset based on user input 'panel_candles_qty'
    // We want the array to be ordered Oldest -> Newest for drawing (left -> right)
    CandleData[] candles_draw = array.new<CandleData>()
    
    for j = panel_candles_qty - 1 to 0
        array.push(candles_draw, array.get(all_candles, j))
    
    // Mapping arrays to store positions
    int[] map_t = array.new_int(panel_candles_qty, na)
    int[] map_x = array.new_int(panel_candles_qty, na)
    
    float panel_top_price = -999999.0
    int center_x_index = 0

        array.set(map_x, i, cx)
